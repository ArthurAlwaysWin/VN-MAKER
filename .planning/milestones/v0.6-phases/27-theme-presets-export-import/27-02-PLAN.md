**Step 2: Create `src/utils/themePackager.js`** — a utility module with file-level JSDoc.

All functions are named exports. Single quotes, 2-space indent, semicolons.

```javascript
/**
 * Theme Packager — ZIP export/import for .theme files.
 *
 * Uses fflate for sync ZIP creation/extraction. Handles base64 ↔ binary
 * conversion for nine-slice images and generates swatch preview thumbnails.
 *
 * @module utils/themePackager
 */
import { zipSync, unzipSync, strToU8, strFromU8 } from 'fflate';
```

**Function 1: `base64ToUint8Array(dataUrl)`**
- Strip the `data:image/...;base64,` prefix by splitting on `','` and taking index [1]
- Decode with `atob(base64)` → build `new Uint8Array(binaryStr.length)` → fill with `charCodeAt`
- Return the Uint8Array

**Function 2: `uint8ArrayToBase64DataUrl(bytes, mimeType = 'image/png')`**
- Build binary string: iterate bytes, `String.fromCharCode(bytes[i])`
- Return `` `data:${mimeType};base64,${btoa(binary)}` ``

**Function 3: `generateSwatchPreview(tokens, width = 320, height = 180)`**
- Create a `document.createElement('canvas')` with given dimensions
- Get 2d context
- Draw background: `ctx.fillStyle = tokens['panel-bg'] || '#0a0a14'; ctx.fillRect(0, 0, width, height)`
- Draw primary color bar at top: `ctx.fillStyle = tokens['primary']; ctx.fillRect(0, 0, width, 40)`
- Draw accent swatch: `tokens['accent']` at (16, 56, 60, 40)
- Draw button sample: `tokens['btn-bg']` at (92, 56, 100, 40)
- Draw card background: `tokens['card-bg']` at (208, 56, 96, 40)
- Draw text samples: `tokens['text']` — `ctx.font = '14px sans-serif'; ctx.fillText('预览文字', 16, 130)`
- Draw secondary text: `tokens['text-secondary']` — `ctx.fillText('次要文字', 16, 150)`
- Draw border-hover line: `ctx.strokeStyle = tokens['border-hover']; ctx.lineWidth = 2; ctx.strokeRect(16, 108, width - 32, 1)`
- Return `canvas.toDataURL('image/png')`

**Function 4: `buildThemeZip(themeData, metadata = {})`**
- `themeData` shape: `{ tokens: {}, nineSlice?: {} }`
- Build the files object for zipSync
- Process nineSlice images: iterate `themeData.nineSlice` entries. For each key where `config.src` starts with `'data:'`:
  - Convert to binary: `files['images/' + key + '.png'] = base64ToUint8Array(config.src)`
  - Store reference path in a `nineSliceRefs` object: `{ ...config, src: 'images/' + key + '.png' }`
  - Also handle button states: if `config.states`, iterate `hover`/`active`, convert each `stateConfig.src` if it starts with `'data:'`, filename: `'images/' + key + '_' + state + '.png'`
- Generate swatch preview: `const previewBase64 = generateSwatchPreview(themeData.tokens || {})`
  - `files['images/preview.png'] = base64ToUint8Array(previewBase64)`
- Build theme.json object:
  ```javascript
  const themeJson = {
    formatVersion: 1,
    name: metadata.name || '自定义主题',
    description: metadata.description || '',
    author: metadata.author || '',
    createdAt: metadata.createdAt || new Date().toISOString(),
    previewImage: 'images/preview.png',
    tokens: themeData.tokens || {},
    nineSlice: hasNineSliceRefs ? nineSliceRefs : undefined,
  };
  ```
- `files['theme.json'] = strToU8(JSON.stringify(themeJson, null, 2))`
- Return `zipSync(files)` — this returns a `Uint8Array`

**Function 5: `parseThemeZip(zipBuffer)`**
- `const unzipped = unzipSync(new Uint8Array(zipBuffer))`
- Parse theme.json: `const themeJson = JSON.parse(strFromU8(unzipped['theme.json']))`
- Validate: check `themeJson.formatVersion` is a number, `themeJson.tokens` is an object. If `formatVersion > 1`, log warning `console.warn('[themePackager] formatVersion', themeJson.formatVersion, '> 1, attempting import')`.
- Reconstruct nineSlice with base64: if `themeJson.nineSlice`, iterate entries. For each config where `config.src` is a string path (not starting with `'data:'`):
  - Look up `unzipped[config.src]` → if exists, convert `uint8ArrayToBase64DataUrl(bytes)` → replace config.src with data URL
  - Same for button states: `config.states.hover.src`, `config.states.active.src`
- Build result theme object:
  ```javascript
  return {
    success: true,
    theme: {
      tokens: themeJson.tokens || {},
      nineSlice: themeJson.nineSlice || undefined,
    },
    metadata: {
      formatVersion: themeJson.formatVersion,
      name: themeJson.name,
      description: themeJson.description,
      author: themeJson.author,
      createdAt: themeJson.createdAt,
    },
  };
  ```
- Wrap in try/catch: on error return `{ success: false, error: e.message }`

**Step 3: Add IPC handlers to `electron/main.js`**

Add 2 new handlers AFTER the existing IPC handlers section (after `ipcMain.handle('save-project', ...)`):

```javascript
// ─── Theme Export/Import IPC ─────────────────────────────

ipcMain.handle('export-theme', async (event, { buffer }) => {
  try {
    const result = await dialog.showSaveDialog(getMainWindow(), {
      title: '导出主题',
      defaultPath: 'my-theme.theme',
      filters: [{ name: '主题文件', extensions: ['theme'] }],
    });
    if (result.canceled) return { success: false, canceled: true };
    await fs.writeFile(result.filePath, Buffer.from(buffer));
    return { success: true, path: result.filePath };
  } catch (e) {
    console.error('[export-theme] Failed:', e);
    return { success: false, error: e.message };
  }
});

ipcMain.handle('import-theme', async () => {
  try {
    const result = await dialog.showOpenDialog(getMainWindow(), {
      title: '导入主题',
      filters: [{ name: '主题文件', extensions: ['theme'] }],
      properties: ['openFile'],
    });
    if (result.canceled) return { success: false, canceled: true };
    const data = await fs.readFile(result.filePaths[0]);
    return { success: true, buffer: data };
  } catch (e) {
    console.error('[import-theme] Failed:', e);
    return { success: false, error: e.message };
  }
});
```

Note: `buffer` parameter from renderer is a Uint8Array (Electron 41 structured clone handles this). On the main process side, `Buffer.from(buffer)` converts for fs.writeFile. On return, `data` from fs.readFile is a Node Buffer which transfers back via structured clone.

**2. Add export handler function:**
```javascript
async function onExport() {
  try {
    exportStatus.value = '导出中...';
    const theme = script.getTheme();
    if (!theme) { exportStatus.value = '无主题数据'; return; }

    const zipBuffer = buildThemeZip(theme, {
      name: '自定义主题',
      description: '',
      author: '',
      createdAt: new Date().toISOString(),
    });

    const result = await window.ipcRenderer.invoke('export-theme', { buffer: zipBuffer });
    if (result.canceled) {
      exportStatus.value = '';
      return;
    }
    if (!result.success) {
      exportStatus.value = '导出失败: ' + result.error;
      return;
    }
    exportStatus.value = '导出成功';
    setTimeout(() => { exportStatus.value = ''; }, 3000);
  } catch (e) {
    console.error('[PresetModal] Export failed:', e);
    exportStatus.value = '导出失败: ' + e.message;
  }
}
```

**3. Add import handler function:**
```javascript
async function onImport() {
  try {
    exportStatus.value = '导入中...';
    const fileResult = await window.ipcRenderer.invoke('import-theme');
    if (fileResult.canceled) {
      exportStatus.value = '';
      return;
    }
    if (!fileResult.success) {
      exportStatus.value = '读取失败: ' + fileResult.error;
      return;
    }

    const parsed = parseThemeZip(fileResult.buffer);
    if (!parsed.success) {
      exportStatus.value = '解析失败: ' + parsed.error;
      return;
    }

    // Apply imported theme — full overwrite per D-12, pushes undo stack
    script.updateTheme(parsed.theme);
    editor.flushPreview();
    exportStatus.value = '导入成功';
    setTimeout(() => {
      exportStatus.value = '';
      emit('close');
    }, 500);
  } catch (e) {
    console.error('[PresetModal] Import failed:', e);
    exportStatus.value = '导入失败: ' + e.message;
  }
}
```
