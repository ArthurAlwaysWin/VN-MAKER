    **Complete file content:**
    ```javascript
    /**
     * vite.web.config.js — Standalone web build configuration.
     *
     * Builds only the runtime engine (index.html → src/main.js) into
     * deterministic engine.js + engine.css output files for web export.
     *
     * Usage: npx vite build --config vite.web.config.js
     * Or:    npm run build:web
     *
     * Per D-03: Independent from vite.config.js (no vite-plugin-electron).
     * Per D-04: Deterministic filenames (no content hash).
     */
    import { defineConfig } from 'vite';
    import { resolve } from 'path';

    export default defineConfig({
      build: {
        outDir: 'dist-web',
        emptyOutDir: true,
        rollupOptions: {
          input: resolve(__dirname, 'index.html'),
          output: {
            entryFileNames: 'engine.js',
            chunkFileNames: 'engine-[name].js',
            assetFileNames: (assetInfo) => {
              if (assetInfo.name?.endsWith('.css')) return 'engine.css';
              return 'assets/[name][extname]';
            },
          },
        },
      },
    });
    ```

    **Critical design points:**
    - NO `vue()` plugin — runtime engine is pure vanilla JS, does NOT import Vue SFCs
    - NO `electron` plugins — excluded per D-03
    - `outDir: 'dist-web'` — avoids conflict with `dist/` (Vite dev) and `dist-electron/` (Electron build)
    - Single entry `resolve(__dirname, 'index.html')` — not multi-entry like vite.config.js
    - `entryFileNames: 'engine.js'` — deterministic JS name per D-04
    - `assetFileNames` function: `.css` files → `engine.css`, other assets → `assets/[name][extname]`
    - `chunkFileNames: 'engine-[name].js'` — deterministic chunk names if code splitting occurs
    - `emptyOutDir: true` — clean build directory each time
    - Uses `__dirname` pattern consistent with existing vite.config.js (Vite injects __dirname for config files)
  </action>
