**Methods (all use `JSON.parse(JSON.stringify())` before IPC per INFRA-04):**

`async function loadCategory(category)`:
- Guard: `if (!window.ipcRenderer) return;`
- Call: `const result = await window.ipcRenderer.invoke('list-assets', JSON.parse(JSON.stringify({ category })));`
- If result.success: `files.value[category] = result.files;`

`async function loadAll()`:
- Set `isLoading.value = true`
- `await Promise.all(['backgrounds', 'characters', 'audio', 'fonts', 'ui'].map(c => loadCategory(c)));`
- Set `isLoading.value = false`

`async function importAssets(category, fileDataArray)`:
- `const cleanData = JSON.parse(JSON.stringify(fileDataArray));`
- `const result = await window.ipcRenderer.invoke('import-assets', { category, files: cleanData });`
- If result.success: `await loadCategory(category);` (refresh file list)
- Return result (caller needs imported/errors for UI feedback)

`async function deleteAsset(category, filename)`:
- `const result = await window.ipcRenderer.invoke('delete-asset', JSON.parse(JSON.stringify({ category, filename })));`
- If result.success: `await loadCategory(category);` (refresh)
- Return result

`async function selectAsset(types)`:
- `const result = await window.ipcRenderer.invoke('select-asset', JSON.parse(JSON.stringify({ types })));`
- Return result (string path or null)

`function syncFontMeta(scriptData)`:
- `fontMeta.value = scriptData?.assets?.fonts || [];`
- Called by App.vue after loading script data to sync font metadata

`async function loadProjectFonts(scriptData)`:
- Call `syncFontMeta(scriptData)` first
- If `fontMeta.value.length === 0` return `{ loaded: [], failed: [] }`
- Call `const result = await loadAllFonts(fontMeta.value, 'asset://');`
- Return result (for corrupt font handling per D-05)

`async function importFont(category, fileDataArray, scriptStore)`:
- Call `const result = await importAssets(category, fileDataArray);`
- If result.success and result.imported.length > 0:
  - For each imported file: create font metadata object:
    ```javascript
    const meta = {
      id: `font-${Date.now()}-${i}`,
      name: path.parse(file.saved).name, // e.g., 'NotoSerifSC' from 'NotoSerifSC.ttf'
      file: `fonts/${file.saved}`,
      family: `UserFont-${path.parse(file.saved).name}`,
    };
    ```
    Note: Use filename parsing since there's no `path` in browser. Extract name: `file.saved.replace(/\.[^.]+$/, '')`.
  - Add to script data: `scriptStore.data.assets ??= {}; scriptStore.data.assets.fonts ??= []; scriptStore.data.assets.fonts.push(meta);`
  - Call `scriptStore.pushState();` to enable undo
  - Call `syncFontMeta(scriptStore.data);`
  - Hot-load via `await loadSingleFont(meta, 'asset://');` per D-06
- Return result

**Return statement:**
```javascript
return {
  files, fontMeta, isLoading, fontFamilies,

  syncFontMeta, loadProjectFonts, importFont,
};
```

3. Modify the `openProject` function (lines 157-166) to add asset loading and font loading after project loads:
```javascript
async function openProject(projectPath) {
  const result = await project.loadProject(projectPath);
  if (result && result.success) {
    script.loadFromData(result.script);

    // Load asset file lists for all categories
    await assets.loadAll();

    // Load custom fonts into editor window (D-04: all fonts on project open)
    const fontResult = await assets.loadProjectFonts(script.data);
    if (fontResult.failed.length > 0) {
      // D-05: Show error for corrupt fonts, ask to delete
      for (const f of fontResult.failed) {
        const msg = `${f.file} 加载失败: ${f.error}`;
        console.error('[FontLoader]', msg);
        if (window.ipcRenderer) {
          const { response } = await window.ipcRenderer.invoke('show-message-box', {
            type: 'warning',
            buttons: ['删除字体文件', '跳过'],
            defaultId: 1,
            cancelId: 1,
            title: '字体加载失败',
            message: `${f.file} 加载失败`,
            detail: `错误: ${f.error}\n\n是否删除此损坏的字体文件？`,
          });
          if (response === 0) {
            await assets.deleteAsset('fonts', f.file.replace('fonts/', ''));
          }
        }
      }
    }

    currentView.value = 'editing';
    activeTab.value = 'scenes';
  } else if (result && result.error) {
    alert(result.error);
  }
}
```
