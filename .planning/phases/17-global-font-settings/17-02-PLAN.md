```javascript
import { computed } from 'vue';
import { useScriptStore } from '../stores/script.js';
import { useAssetStore } from '../stores/assets.js';

const script = useScriptStore();
const assets = useAssetStore();

const systemFonts = [
  { label: 'Noto Sans SC', value: "'Noto Sans SC', sans-serif" },
  { label: 'Noto Serif SC', value: "'Noto Serif SC', serif" },
  { label: 'Sans Serif', value: 'sans-serif' },
  { label: 'Serif', value: 'serif' },
  { label: 'Monospace', value: 'monospace' },
];

const settings = computed(() => script.getDialogueBox());

function setField(field, value) {
  if (!settings.value) return;
  const updated = { ...settings.value, [field]: value };
  script.updateDialogueBox(updated);
}

// Slider: immediate visual feedback on @input (no pushState),
// commit on @change (pushState via updateDialogueBox) — per Research Pitfall 5
function onSliderInput(field, event) {
  if (!settings.value) return;
  settings.value[field] = Number(event.target.value);
  // Direct mutation for instant canvas reactivity — no pushState
}

function onSliderCommit(field, event) {
  if (!settings.value) return;
  const updated = { ...settings.value, [field]: Number(event.target.value) };
  script.updateDialogueBox(updated);
}

// Mini preview computed styles
const previewTextStyle = computed(() => ({
  fontSize: (settings.value?.fontSize ?? 18) + 'px',
  fontFamily: settings.value?.fontFamily || "'Noto Sans SC', sans-serif",
  color: settings.value?.textColor || 'rgba(255, 255, 255, 0.92)',
  lineHeight: '1.8',
}));

const previewNameplateStyle = computed(() => ({
  fontSize: (settings.value?.nameplateFontSize ?? 20) + 'px',
  fontFamily: settings.value?.nameplateFontFamily || "'Noto Serif SC', serif",
  color: settings.value?.nameplateColor || '#ffd700',
  fontWeight: '600',
  letterSpacing: '2px',
  marginBottom: '8px',
}));
```

```javascript
function getDialogueStyle(dlg) {
  const globalFont = script.data?.ui?.dialogueBox;
  const pageOverride = props.currentPage?.fontOverride;
  const useOverride = pageOverride && !pageOverride.useGlobal;

  const s = {
    width: '100%',
    height: '100%',
    padding: '15px 20px',
    background: 'rgba(0, 0, 0, 0.7)',
    borderRadius: '8px',
    color: '#fff',
    fontFamily: "'Noto Sans SC', sans-serif",
    fontSize: '16px',
    overflow: 'hidden',
  };

  // Apply global font settings as base
  if (globalFont) {
    if (globalFont.fontSize) s.fontSize = globalFont.fontSize + 'px';
    if (globalFont.textColor) s.color = globalFont.textColor;
    if (globalFont.fontFamily) s.fontFamily = globalFont.fontFamily;
  }

  // Apply per-page override if active (per D-04)
  if (useOverride) {
    if (pageOverride.fontSize) s.fontSize = pageOverride.fontSize + 'px';
    if (pageOverride.textColor) s.color = pageOverride.textColor;
    if (pageOverride.fontFamily) s.fontFamily = pageOverride.fontFamily;
  }

  // Per-dialogue style overrides still take highest priority
  if (dlg.style) {
    if (dlg.style.fontSize) s.fontSize = dlg.style.fontSize + 'px';
    if (dlg.style.fontFamily) s.fontFamily = dlg.style.fontFamily;
    if (dlg.style.textColor) s.color = dlg.style.textColor;
    if (dlg.style.backgroundColor) s.background = dlg.style.backgroundColor;
    if (dlg.style.borderRadius != null) s.borderRadius = dlg.style.borderRadius + 'px';
    if (dlg.style.padding != null) s.padding = dlg.style.padding + 'px';
  }
  return s;
}
```

const fontOverride = computed(() => {
  if (!page.value) return null;
  return page.value.fontOverride || null;
});

const useGlobalFont = computed(() => {
  return !fontOverride.value || fontOverride.value.useGlobal !== false;
});

function toggleUseGlobal(checked) {
  if (!page.value) return;
  if (checked) {
    // Switch to global — set useGlobal: true (keep existing override values)
    if (page.value.fontOverride) {
      page.value.fontOverride.useGlobal = true;
    }
  } else {
    // Switch to per-page override — initialize from global defaults
    const global = script.getDialogueBox() || {};
    page.value.fontOverride = {
      useGlobal: false,
      fontSize: global.fontSize ?? 18,
      fontFamily: global.fontFamily ?? null,
      textColor: global.textColor ?? null,
      nameplateFontSize: global.nameplateFontSize ?? 20,
      nameplateFontFamily: global.nameplateFontFamily ?? null,
      nameplateColor: global.nameplateColor ?? null,
    };
  }
  script.pushState();
}

function setOverrideField(field, value) {
  if (!page.value?.fontOverride) return;
  page.value.fontOverride[field] = value;
  // Discrete change (select, color) — commit immediately
  script.pushState();
}

function onOverrideSliderInput(field, event) {
  if (!page.value?.fontOverride) return;
  page.value.fontOverride[field] = Number(event.target.value);
  // Continuous slider — no pushState (Pitfall 5)
}

function onOverrideSliderCommit(field, event) {
  if (!page.value?.fontOverride) return;
  page.value.fontOverride[field] = Number(event.target.value);
  script.pushState();
}
