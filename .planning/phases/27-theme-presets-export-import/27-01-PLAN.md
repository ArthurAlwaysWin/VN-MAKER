**Step 1: Create `tools/generatePresets.js`** — a Node.js ESM script that generates preset token values.

The script imports from the engine modules (relative paths from tools/):
```javascript
import { generatePalette, hexToHsl, hslToHex } from '../src/engine/colorHarmony.js';
import { contrastRatio, autoFix } from '../src/engine/contrast.js';
import { DEFAULT_TOKENS } from '../src/engine/tokens.js';
```

For each of the 4 presets, the script:
1. Calls `generatePalette(primaryHex, algorithm)` to get 34 hex color tokens
2. Applies alpha preservation from DEFAULT_TOKENS — for each token key, if DEFAULT_TOKENS[key] starts with `'rgba'`, extract its alpha via regex `rgba?\(\s*\d+\s*,\s*\d+\s*,\s*\d+\s*,\s*([\d.]+)\)` and wrap the generated hex color as `rgba(r, g, b, alpha)`. If DEFAULT_TOKENS[key] starts with `'linear-gradient'`, generate a gradient using the preset's dark background hex.
3. Overrides `danger` to `'#ff6b6b'` and `danger-hover` to `'rgba(255, 100, 100, 0.9)'` for ALL presets (red danger color is a UX convention, not theme-dependent).
4. Keeps non-color tokens from DEFAULT_TOKENS as fallback or customizes per preset:
   - `font-body`: keep DEFAULT_TOKENS value for all presets
   - `font-display`: keep DEFAULT_TOKENS value for all presets
   - `radius`: '4px' for Modern/Minimal, '2px' for Japanese, '6px' for Fantasy
   - `radius-lg`: '6px' for Modern/Minimal, '4px' for Japanese, '8px' for Fantasy
   - `blur`: '8px' for all
5. Runs `contrastRatio(textHex, panelBgHex)` for the `text` token vs `panel-bg` token. If ratio < 4.5, calls `autoFix(textHex, panelBgHex, 4.5)` and uses the fixed hex. Same check for `btn-text` vs `btn-bg`.
6. Outputs the 4 preset objects as JSON to stdout.

The 4 presets and their algorithms (per D-04):
- `modern`: primaryColor `'#4A90D9'`, algorithm `'complementary'`, name `'现代'`, description `'清新蓝色调，适合现代都市题材'`
- `japanese`: primaryColor `'#C8A882'`, algorithm `'analogous'`, name `'和风'`, description `'温暖大地色调，适合日本传统题材'`
- `fantasy`: primaryColor `'#7B2FBE'`, algorithm `'triadic'`, name `'幻想'`, description `'神秘紫色调，适合奇幻冒险题材'`
- `minimal`: primaryColor `'#333333'`, algorithm `'complementary'`, name `'简约'`, description `'低饱和度灰色调，适合简洁现代题材'`

Add these methods after the existing `resetTheme()`:

```javascript
function previewPreset(presetTokens) {
  // Immediately send preview to iframe — no debounce, no store write (D-02)
  if (!iframeRef.value?.contentWindow || !isEngineReady.value) return;
  const theme = script.getTheme();
  iframeRef.value.contentWindow.postMessage({
    type: 'update-theme',
    theme: {
      tokens: { ...presetTokens },
      nineSlice: theme?.nineSlice ?? {},
    },
  }, '*');
}

function applyPreset(presetTokens) {
  // Write to store + push undo (D-02, D-07, PRE-02)
  const theme = script.getTheme();
  const clone = JSON.parse(JSON.stringify({
    tokens: { ...presetTokens },
    nineSlice: theme?.nineSlice ?? {},
  }));
  script.updateTheme(clone);
  flushPreview();
}

function cancelPreview() {
  // Restore iframe to actual store state (Pitfall 5)
  flushPreview();
}
```

Script setup:
```javascript
import { ref } from 'vue';
import { useThemeEditor } from '../../composables/useThemeEditor.js';
import { THEME_PRESETS } from '../../../engine/presets.js';

const emit = defineEmits(['close']);
const editor = useThemeEditor();
const selectedId = ref(null);

function onSelectPreset(preset) {
  selectedId.value = preset.id;
  editor.previewPreset(preset.tokens);
}

function onApply() {
  const preset = THEME_PRESETS.find(p => p.id === selectedId.value);
  if (!preset) return;
  editor.applyPreset(preset.tokens);
  emit('close');
}

function onClose() {
  editor.cancelPreview();
  emit('close');
}
```
