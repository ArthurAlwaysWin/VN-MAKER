<feature>
  <name>scanAssets — Pure asset reference extractor</name>
  <files>src/engine/scanAssets.js, tests/scanAssets.test.js</files>
  <behavior>
    ## Input
    Full script.json object containing characters, scenes, assets, and ui sections.

    ## Output (per D-01)
    `{ backgrounds: string[], audio: string[], fonts: string[], characters: string[], voices: string[] }`

    ## Behaviors to test

    ### Return shape
    - Returns object with exactly 5 keys: backgrounds, audio, fonts, characters, voices
    - Each value is an array of strings

    ### Character expressions (→ characters[])
    - characters.hero.expressions.normal = "characters/hero_normal.png" → characters[]
    - Multiple characters, multiple expressions → all included

    ### Scene page assets
    - page.background = "backgrounds/city.png" → backgrounds[]
    - page.bgm.file = "audio/bgm1.mp3" → audio[]
    - page.se.file = "audio/click.mp3" → audio[]
    - page.dialogues[0].voice = "audio/voice01.ogg" → voices[]
    - Dialogue without voice field → nothing added, no error

    ### Fonts (→ fonts[])
    - assets.fonts[0].file = "fonts/myfont.ttf" → fonts[]

    ### UI screens (→ backgrounds[] and audio[])
    - ui.titleScreen.background → backgrounds[]
    - ui.titleScreen.bgm (bare string, NOT object!) → audio[]
    - ui.titleScreen.elements[type='image'].src → backgrounds[]
    - ui.settingsScreen.background → backgrounds[]
    - ui.settingsScreen.elements[type='image'].src → backgrounds[]
    - Non-image elements (type='button', type='slider') → ignored

    ### Filtering
    - data: URI → skipped
    - http:// URL → skipped
    - https:// URL → skipped
    - Empty string / null / undefined → skipped

    ### Deduplication and sorting
    - Same path in multiple pages → appears only once
    - Output arrays sorted alphabetically

    ### Graceful handling
    - Empty script {} → returns 5 empty arrays
    - Missing sections (no characters, no scenes, no ui, no assets) → empty arrays
  </behavior>
  <implementation>
    Config-table-driven traversal using Set for dedup, sorted array output.
    Helper function _add(set, path) filters empty/data:/http:/https: values.
    Walk each of the 11 path locations explicitly — not a recursive tree walker.
  </implementation>
</feature>

    ```javascript
    /**
     * scanAssets — Extract all referenced asset paths from a script object.
     *
     * Pure function, no filesystem dependency. Receives the full script.json
     * data object and returns a categorized dictionary of deduplicated,
     * sorted relative paths suitable for export.
     *
     * Covers 11 path locations across 5 categories (per D-01):
     *   backgrounds — page backgrounds, UI screen backgrounds, UI image elements
     *   audio       — page BGM, page SE, title screen BGM
     *   fonts       — assets.fonts[].file
     *   characters  — characters[id].expressions values
     *   voices      — page dialogues[].voice
     *
     * @module scanAssets
     */

    // ─── Path Filter ─────────────────────────────────────────

    /**
     * Add path to set if it's a valid asset file reference.
     * Skips empty values, data: URIs, and http/https URLs.
     * @param {Set<string>} set - Target set to add to
     * @param {*} path - Raw path value (may be null, undefined, non-string)
     * @private
     */
    function _add(set, path) {
      if (path && typeof path === 'string'
          && !path.startsWith('data:')
          && !path.startsWith('http://')
          && !path.startsWith('https://')) {
        set.add(path);
      }
    }

    // ─── Scanner ─────────────────────────────────────────────

    /**
     * Scan a script object and extract all referenced asset paths.
     *
     * @param {Object} script - The full script.json data object
     * @returns {{ backgrounds: string[], audio: string[], fonts: string[], characters: string[], voices: string[] }}
     */
    export function scanAssets(script) {
      const bg = new Set();
      const audio = new Set();
      const fonts = new Set();
      const chars = new Set();
      const voices = new Set();

      // 1. Character expression images
      for (const char of Object.values(script.characters || {})) {
        for (const imgPath of Object.values(char.expressions || {})) {
          _add(chars, imgPath);
        }
      }

      // 2. Scene pages
      for (const scene of Object.values(script.scenes || {})) {
        for (const page of (scene.pages || [])) {
          _add(bg, page.background);
          if (page.bgm?.file) _add(audio, page.bgm.file);
          if (page.se?.file) _add(audio, page.se.file);
          for (const dlg of (page.dialogues || [])) {
            if (dlg.voice) _add(voices, dlg.voice);
          }
        }
      }

      // 3. Fonts
      for (const font of (script.assets?.fonts || [])) {
        if (font.file) _add(fonts, font.file);
      }

      // 4. UI screens (titleScreen, settingsScreen)
      for (const screenKey of ['titleScreen', 'settingsScreen']) {
        const screen = script.ui?.[screenKey];
        if (!screen) continue;
        _add(bg, screen.background);
        // titleScreen.bgm is a bare string path (not { file: ... } like page bgm)
        if (screenKey === 'titleScreen' && screen.bgm) {
          _add(audio, screen.bgm);
        }
        for (const elem of (screen.elements || [])) {
          if (elem.type === 'image' && elem.src) {
            _add(bg, elem.src);
          }
        }
      }

      // Convert Sets → sorted arrays for deterministic output
      return {
        backgrounds: [...bg].sort(),
        audio: [...audio].sort(),
        fonts: [...fonts].sort(),
        characters: [...chars].sort(),
        voices: [...voices].sort(),
      };
    }
    ```
