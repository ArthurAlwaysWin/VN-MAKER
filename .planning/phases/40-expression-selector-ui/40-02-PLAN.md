<interfaces>
<!-- ExpressionDropdown API (created in Plan 01) -->

Props:
- expressions: Object — { exprName: imagePath } dictionary
- modelValue: String — current expression name ('' for unchanged)
- nullable: Boolean — shows "不變" option when true (default: false)

Emits:
- update:modelValue: String — selected expression name, or '' for unchanged

Usage pattern:
```html
<!-- Required mode (character row) -->
<ExpressionDropdown
  :expressions="getCharExpressions(char.id)"
  :modelValue="char.expression"
  @update:modelValue="setCharExpression(idx, $event)"
  @click.stop
/>

<!-- Nullable mode (dialogue row) -->
<ExpressionDropdown
  :expressions="getCharExpressions(selectedDialogue.speaker)"
  :modelValue="selectedDialogue.expression || ''"
  @update:modelValue="setDialogueExpression($event)"
  nullable
/>
```
</interfaces>
