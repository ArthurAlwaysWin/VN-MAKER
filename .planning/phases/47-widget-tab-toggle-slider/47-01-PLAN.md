**1a. Extend useWidgetStylesEditor.js — add 2 helper methods**

Add an import at the top:

Add these 2 functions before the `// ─── Cleanup ───` section, under a new `// ─── Widget Field Helpers ─────────────────────` divider:

```javascript
/**
 * Write a single field on a widget category without undo push.
 * Used for live preview during continuous interaction (e.g. color picker drag).
 * @param {string} category - 'tab' | 'toggle' | 'slider' | 'panel' | 'button'
 * @param {string} field - field key within the category
 * @param {*} value - new value
 */
function setWidgetField(category, field, value) {
  const ws = script.getWidgetStyles();
  if (!ws) return;
  ws[category] ??= {};
  ws[category][field] = value;
  sendWidgetStylesToPreview();
}

/**
 * Clone current widgetStyles, write to store with undo push, and flush preview.
 * Call after interaction completes (click, mouseup, blur).
 */
function commitWidgetStyles() {
  const ws = script.getWidgetStyles();
  if (!ws) return;
  script.updateWidgetStyles(JSON.parse(JSON.stringify(ws)));
  flushPreview();
}
```

Add `WIDGET_DEFAULTS`, `setWidgetField`, and `commitWidgetStyles` to the `editor` object that gets provided.

**IMPORTANT — rgba-to-hex conversion**: The default colors are in `rgba(...)` format. The `<input type="color">` requires `#rrggbb` hex. Add a helper `rgbaToHex(rgba)` that parses the rgba string and returns hex (same pattern as ColorTokenRow.vue's `parseRgba`). Create computed hex values:
```javascript
function rgbaToHex(color) {
  if (!color) return '#000000';
  if (color.startsWith('#')) return color;
  const m = color.match(/rgba?\(\s*(\d+)\s*,\s*(\d+)\s*,\s*(\d+)/);
  if (!m) return '#000000';
  return '#' + [m[1], m[2], m[3]].map(n => parseInt(n).toString(16).padStart(2, '0')).join('');
}
const trackColorHex = computed(() => rgbaToHex(trackColor.value));
const fillColorHex = computed(() => rgbaToHex(fillColor.value));
const thumbColorHex = computed(() => rgbaToHex(thumbColor.value));
```
